	 public void  writePlayers(){
			FileWriter fw = null;
			BufferedWriter bw=null;
			
		 
			
			File score = new File("src/score.csv");
			if(score.exists()){
				//System.out.println("FIle exists");
			}
			else{
				try {
					score.createNewFile();
					//System.out.println("File created");
				} catch (IOException e) {
					e.printStackTrace();
				}
			}
			
			try{
				fw=new FileWriter(score);
				bw=new BufferedWriter(fw);
				 
				bw.write(" Player,Score,"+Main.ListOfPlayers.players.size()+"\n");
				
				for(int i=0;i<Main.ListOfPlayers.players.size();i++){
                 String name=Main.ListOfPlayers.players.get(i).name;
                 int coins=Main.ListOfPlayers.players.get(i).coins;
                 bw.write(name+","+coins+"\n");
				}
			}catch(Exception e){
				System.out.println("Exception caught 3");
				e.printStackTrace();
			}finally{
				if(bw!=null)
					try {
						bw.close();
					} catch (IOException e) {
						System.out.println("Exception caught 4");
						e.printStackTrace();
					}
			} 
	 }
